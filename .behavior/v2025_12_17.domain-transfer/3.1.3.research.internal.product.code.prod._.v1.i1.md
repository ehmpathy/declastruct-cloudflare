```ts
// detect if domain is registered elsewhere
const whoisRecord = await getOneDomainWhoisRecord({ domain: name }, context);

if (whoisRecord?.found) {
  // domain registered elsewhere → transfer guidance
  BadRequestError.throw(
    `domain "${name}" not found in cloudflare registrar

⚠️ transfer required
   │
   ├─ at ${whoisRecord.registrar ?? 'current registrar'}
   │  └─ steps
   │     ├─ 1. disable dnssec
   │     ├─ 2. disable whois privacy
   │     ├─ 3. unlock domain
   │     └─ 4. get auth code
   │
   └─ at cloudflare
      └─ steps
         ├─ 5. enter domain + auth code
         ├─ 6. confirm contact
         └─ 7. complete transfer`,

  `domain "${name}" not found in cloudflare registrar

⚠️ purchase required
   │
   └─ at cloudflare
      └─ steps
         ├─ 1. search for "${name}"
         ├─ 2. add to cart
         └─ 3. complete purchase`,
  { domain: name },
);
```

## pattern.10: whois detection

### location

src/domain.operations/domainWhoisRecord/getOneDomainWhoisRecord.ts

### pattern

```ts
export const getOneDomainWhoisRecord = async (
  input: { domain: string },
  context: ContextCloudflareApi & ContextLogTrail,
): Promise<DomainWhoisRecord | null> => {
  const response = await context.cloudflare.client.intel.whois.get({
    account_id: context.cloudflare.accountId,
    domain: input.domain,
  });

  if (!response.found) return null;

  return {
    found: response.found,
    registrar: response.registrar ?? null,
    nameservers: response.nameservers ?? [],
  };
};
```

### assessment

| aspect | verdict |
|--------|---------|
| uses cloudflare intel/whois api | ✓ |
| returns null for not found | ✓ |
| extracts registrar for guidance | ✓ |

### decision

**[REUSE]** — pattern is correct, enables transfer vs purchase detection.
