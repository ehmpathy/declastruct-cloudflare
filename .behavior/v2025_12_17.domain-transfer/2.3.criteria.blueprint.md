## subcomponent contracts

### domain objects

given('DeclaredCloudflareDomainZone contract')
  then('extends DomainEntity')
  then('declares static unique = ["name"]')
  then('declares static primary = ["id"]')
  then('exposes: name, type, id?')

given('DeclaredCloudflareDomainDnsRecord contract')
  then('extends DomainEntity')
  then('declares static unique = ["zone.name", "name", "type", "content"]')
  then('declares static primary = ["id"]')
  then('exposes: zone (RefByUnique), name, type, content, ttl, proxied?, priority?')

given('DeclaredCloudflareDomainRegistration contract')
  then('extends DomainEntity')
  then('declares static unique = ["name"]')
  then('exposes: name, autoRenew, locked, privacyProtection')

### provider

given('getDeclastructCloudflareProvider contract')
  then('accepts: { apiToken, accountId }')
  then('returns: DeclastructProvider<CloudflareResourceClass>')
  then('validates credentials at init time')
  then('throws AuthenticationError for invalid credentials')

### zone operations

given('getAllDomainZones contract')
  then('returns: DeclaredCloudflareDomainZone[]')
  then('returns empty array when no zones exist')

given('setDomainZone contract')
  then('accepts: { findsert } | { upsert }')
  then('returns: HasMetadata<DeclaredCloudflareDomainZone>')
  then('creates zone when absent (findsert or upsert)')
  then('returns extant zone without change (findsert)')
  then('updates extant zone with new settings (upsert)')

given('delDomainZone contract')
  then('accepts: { by: { primary } | { unique } }')
  then('removes zone from cloudflare when exists')
  then('succeeds silently when zone absent (idempotent)')

### dns record operations

given('getAllDomainDnsRecords contract')
  then('accepts: { by: { zone: Ref<Zone> } }')
  then('returns: DeclaredCloudflareDomainDnsRecord[]')
  then('returns empty array when no records exist')

given('setDomainDnsRecord contract')
  then('accepts: { findsert } | { upsert }')
  then('returns: HasMetadata<DeclaredCloudflareDomainDnsRecord>')
  then('creates record when absent')
  then('returns extant record without change (findsert)')
  then('updates extant record with new values (upsert)')
  then('allows multiple records with same (zone, name, type) for load distribution')

given('delDomainDnsRecord contract')
  then('accepts: { by: { primary } | { unique } }')
  then('removes record from cloudflare when exists')
  then('succeeds silently when record absent (idempotent)')

### registration operations

given('getAllDomainRegistrations contract')
  then('returns: DeclaredCloudflareDomainRegistration[]')
  then('lists all domains registered with cloudflare registrar')

given('setDomainRegistration contract')
  then('accepts: { findsert } | { upsert }')
  then('returns: HasMetadata<DeclaredCloudflareDomainRegistration>')
  then('updates settings when domain is in cloudflare registrar')
  then('throws BadRequestError with transfer guidance when domain is elsewhere')
  then('throws BadRequestError with purchase guidance when domain is unregistered')

### error contracts

given('BadRequestError contract')
  then('extends HelpfulError')
  then('includes structured message with treestruct guidance')
  then('includes metadata: { domain, registrar?, accountId }')

given('AuthenticationError contract')
  then('extends HelpfulError')
  then('surfaces clear message about credential failure')

## composition boundaries

### provider composes operations

given('DeclastructCloudflareProvider implementation')
  then('composes: cloudflare sdk client')
  then('composes: zone operations')
  then('composes: dns record operations')
  then('composes: registration operations')
  then('exposes: dao interface for declastruct core')

### registration guidance flow

given('setDomainRegistration guidance flow')
  then('queries cloudflare registrar for domain')
  then('if found in cloudflare: findsert/upsert as normal')
  then('if not found: queries WHOIS to determine domain state')
  then('if WHOIS shows registered elsewhere: throws transfer guidance')
  then('if WHOIS shows unregistered: throws purchase guidance')

### declastruct integration

given('declastruct plan/apply integration')
  then('provider exposes: DeclastructDao interface')
  then('domain objects expose: static unique, static primary')
  then('declastruct core handles: dependency resolution (zones before records)')
  then('declastruct core handles: diff computation')
  then('declastruct core handles: plan serialization')

## integration boundaries

given('cloudflare api boundary')
  then('all api calls via official cloudflare sdk')
  then('authentication via api token (not api key)')
  then('account id required for account-scoped operations')

given('declastruct core boundary')
  then('provider implements DeclastructProvider interface')
  then('dao methods follow findsert/upsert/del pattern')
  then('domain objects follow domain-objects conventions')

given('WHOIS api boundary')
  then('uses cloudflare /accounts/{account_id}/intel/whois endpoint')
  then('returns registrar info when domain registered elsewhere')
  then('returns empty when domain unregistered')
