## pattern.3: sample context factory

```ts
export const getSampleCloudflareApiContext = (): ContextCloudflareApi &
  ContextLogTrail => {
  // fail-fast: require env vars
  const apiToken = process.env.CLOUDFLARE_API_TOKEN;
  const accountId = process.env.CLOUDFLARE_ACCOUNT_ID;

  if (!apiToken)
    BadRequestError.throw(
      'CLOUDFLARE_API_TOKEN env var required for integration tests',
    );
  if (!accountId)
    BadRequestError.throw(
      'CLOUDFLARE_ACCOUNT_ID env var required for integration tests',
    );

  // create the client
  const client = new Cloudflare({ apiToken });

  return {
    cloudflare: { client, accountId },
    log: { debug: () => {}, info: () => {}, warn: () => {}, error: () => {} },
  };
};
```

### assessment

| aspect | verdict |
|--------|---------|
| creates real cloudflare client | ✓ |
| fails fast on absent credentials | ✓ |
| provides clear error messages | ✓ |

### decision

## pattern.12: resources file for acceptance tests

### location

src/contract/sdks/.test/assets/resources.acceptance.ts

### pattern

```ts
import { refByUnique } from 'domain-objects';
import { getDeclastructCloudflareProvider } from '../../../../../dist/contract/sdks';

export const getProviders = async () => [
  await getDeclastructCloudflareProvider(
    { apiToken: process.env.CLOUDFLARE_API_TOKEN!, accountId: process.env.CLOUDFLARE_ACCOUNT_ID! },
    { log: console },
  ),
];

export const getResources = async () => {
  const testZoneName = process.env.CLOUDFLARE_TEST_ZONE_NAME;
  if (!testZoneName) return [];

  const zone = new DeclaredCloudflareDomainZone({ name: testZoneName, type: 'full' });
  const txtRecord = new DeclaredCloudflareDomainDnsRecord({
    zone: refByUnique(zone),
    name: `declastruct-acceptance-test.${testZoneName}`,
    type: 'TXT',
    content: `declastruct-test-${Date.now()}`,
    ttl: 1,
    proxied: false,
  });

  return [zone, txtRecord];
};
```

### assessment

| aspect | verdict |

| imports from dist/ (published package) | ✓ |
| exports getProviders and getResources | ✓ |
| supports empty resources mode | ✓ |

### decision
