## 1. usecases

### usecase: manage cloudflare zones

```typescript
import { PaginationInput } from 'pagination-fns';

// list all zones in account (via domain operation, not dao)
const zones = await getAllDomainZones({
  page: { limit: 50 },
}, context);

// get a specific zone by name (via dao)
const zone = await DeclaredCloudflareDomainZoneDao.get.one.byUnique({
  name: 'example.com',
}, context);

// get a specific zone by id (via dao)
const zone = await DeclaredCloudflareDomainZoneDao.get.one.byPrimary({
  id: 'zone-id-123',
}, context);

// create or find a zone (via dao)
const zone = await DeclaredCloudflareDomainZoneDao.set.findsert({
  name: 'example.com',
  type: 'full',
}, context);

// delete a zone (via dao)
await DeclaredCloudflareDomainZoneDao.set.delete({
  by: { name: 'example.com' },
}, context);
```

### usecase: manage dns records

```typescript
import { Ref } from 'domain-objects';
import { PaginationInput } from 'pagination-fns';

// list all records in a zone (via domain operation, not dao)
const records = await getAllDomainDnsRecords({
  by: { zone: { name: 'example.com' } },  // Ref by unique
  page: { limit: 100 },
}, context);

// list all records using zone id (via domain operation)
const records = await getAllDomainDnsRecords({
  by: { zone: { id: 'zone-id-123' } },  // Ref by primary
  page: { limit: 100 },
}, context);

// get a specific record by unique key (via dao)
const record = await DeclaredCloudflareDomainDnsRecordDao.get.one.byUnique({
  zone: { name: 'example.com' },
  name: 'api.example.com',
  type: 'A',
}, context);

// get a specific record by id (via dao)
const record = await DeclaredCloudflareDomainDnsRecordDao.get.one.byPrimary({
  id: 'record-id-456',
}, context);

// upsert a dns record (zone ref enables declaration before zone exists)
const record = await DeclaredCloudflareDomainDnsRecordDao.set.upsert({
  zone: { name: 'example.com' },  // Ref - resolved at runtime
  name: 'api.example.com',
  type: 'A',
  content: '192.168.1.1',
  ttl: 300,
  proxied: true,
}, context);

// delete a dns record (via dao)
await DeclaredCloudflareDomainDnsRecordDao.set.delete({
  by: {
    zone: { name: 'example.com' },
    name: 'api.example.com',
    type: 'A',
  },
}, context);
```

## 2. domain.objects

### 2.1 entities

#### DeclaredCloudflareDomainZone

```typescript
/**
 * .what = a cloudflare zone (domain container for dns management)
 * .why = enables declarative dns and registrar management
 *
 * .identity
 *   - @primary = [id] — assigned by cloudflare on creation
 *   - @unique = [name] — domain names are unique within an account
 */
export interface DeclaredCloudflareDomainZone {
  /**
   * .what = the cloudflare zone id
   * .note = @metadata — assigned by cloudflare on creation
   */
  id?: string;

  /**
   * .what = the domain name
   * .note = @unique — domain names are unique within an account
   * .example = 'example.com'
   */
  name: string;

  /**
   * .what = the zone type
   * .note = must be 'full' for registrar features
   */
  type: 'full' | 'partial' | 'secondary';

  /**
   * .what = whether the zone is paused
   */
  paused?: boolean;

  /**
   * .what = the zone status
   * .note = @readonly — managed by cloudflare
   */
  status?: 'initializing' | 'pending' | 'active' | 'moved';

  /**
   * .what = cloudflare-assigned nameservers
   * .note = @readonly — assigned by cloudflare
   */
  nameServers?: string[];

  /**
   * .what = original nameservers before cloudflare
   * .note = @readonly — captured on zone creation
   */
  originalNameservers?: string[];

  /**
   * .what = original registrar name
   * .note = @readonly
   */
  originalRegistrar?: string;

  /**
   * .what = when the zone was created
   * .note = @readonly
   */
  createdOn?: string;

  /**
   * .what = when the zone was activated
   * .note = @readonly
   */
  activatedOn?: string;
}

export class DeclaredCloudflareDomainZone
  extends DomainEntity<DeclaredCloudflareDomainZone>
  implements DeclaredCloudflareDomainZone
{
  public static primary = ['id'] as const;
  public static unique = ['name'] as const;
  public static metadata = ['id'] as const;
  public static readonly = [
    'status',
    'nameServers',
    'originalNameservers',
    'originalRegistrar',
    'createdOn',
    'activatedOn',
  ] as const;
}
```

#### DeclaredCloudflareDomainDnsRecord

```typescript
import { Ref } from 'domain-objects';

/**
 * .what = a cloudflare dns record
 * .why = enables declarative dns record management
 *
 * .identity
 *   - @primary = [id] — assigned by cloudflare on creation
 *   - @unique = [zone, name, type] — composite unique within zone
 *
 * .note
 *   - same name can have multiple records of different types
 *   - A/AAAA cannot coexist with CNAME on same name
 *   - zone is a Ref, enabling declaration before zone exists
 */
export interface DeclaredCloudflareDomainDnsRecord {
  /**
   * .what = the record id
   * .note = @metadata — assigned by cloudflare
   */
  id?: string;

  /**
   * .what = the parent zone reference
   * .note = @unique (part of composite key)
   * .why = enables declaring records for zones that don't exist yet
   */
  zone: Ref<typeof DeclaredCloudflareDomainZone>;

  /**
   * .what = the dns record name
   * .note = @unique (part of composite key)
   * .example = 'api.example.com'
   */
  name: string;

  /**
   * .what = the record type
   * .note = @unique (part of composite key)
   */
  type: DeclaredCloudflareDomainDnsRecordType;

  /**
   * .what = the record content/value
   */
  content: string;

  /**
   * .what = time-to-live in seconds
   * .note = 1 = automatic, 60-86400 for manual (30+ for enterprise)
   */
  ttl: number;

  /**
   * .what = whether cloudflare proxies traffic
   */
  proxied?: boolean;

  /**
   * .what = optional comment/annotation
   */
  comment?: string;

  /**
   * .what = optional tags
   */
  tags?: string[];

  /**
   * .what = priority for MX records
   * .note = only applicable to MX type
   */
  priority?: number;

  /**
   * .what = record-specific settings
   */
  settings?: DeclaredCloudflareDomainDnsRecordSettings;

  /**
   * .what = when the record was created
   * .note = @readonly
   */
  createdOn?: string;

  /**
   * .what = when the record was last modified
   * .note = @readonly
   */
  modifiedOn?: string;

  /**
   * .what = whether the record can be proxied
   * .note = @readonly
   */
  proxiable?: boolean;
}

export class DeclaredCloudflareDomainDnsRecord
  extends DomainEntity<DeclaredCloudflareDomainDnsRecord>
  implements DeclaredCloudflareDomainDnsRecord
{
  public static primary = ['id'] as const;
  public static unique = ['zone', 'name', 'type'] as const;
  public static metadata = ['id'] as const;
  public static readonly = ['createdOn', 'modifiedOn', 'proxiable'] as const;
  public static nested = {
    zone: DeclaredCloudflareDomainZone,
    settings: DeclaredCloudflareDomainDnsRecordSettings,
  };
}
```

#### DeclaredCloudflareDomainRegistration

```typescript
/**
 * .what = a domain registration managed via cloudflare registrar
 * .why = enables declarative domain settings management
 *
 * .identity
 *   - @primary = [id] — assigned by cloudflare
 *   - @unique = [name] — domain names are globally unique
 *
 * .note
 *   - registration and transfer cannot be automated via api
 *   - only settings (auto_renew, privacy, locked) can be set via api
 */
export interface DeclaredCloudflareDomainRegistration {
  /**
   * .what = the cloudflare domain id
   * .note = @metadata — assigned by cloudflare
   */
  id?: string;

  /**
   * .what = the domain name
   * .note = @unique
   */
  name: string;

  /**
   * .what = whether auto-renewal is enabled
   */
  autoRenew: boolean;

  /**
   * .what = whether whois privacy is enabled
   */
  privacy: boolean;

  /**
   * .what = whether the domain is locked
   * .note = locked domains cannot be transferred
   */
  locked: boolean;

  /**
   * .what = whether the domain is available for registration
   * .note = @readonly
   */
  available?: boolean;

  /**
   * .what = whether the domain can be registered
   * .note = @readonly
   */
  canRegister?: boolean;

  /**
   * .what = the current registrar name
   * .note = @readonly
   */
  currentRegistrar?: string;

  /**
   * .what = when the domain expires
   * .note = @readonly
   */
  expiresAt?: string;

  /**
   * .what = when the domain was registered
   * .note = @readonly
   */
  createdAt?: string;

  /**
   * .what = registry status codes
   * .note = @readonly
   */
  registryStatuses?: string;

  /**
   * .what = whether the tld is supported
   * .note = @readonly
   */
  supportedTld?: boolean;

  /**
   * .what = registrant contact information
   * .note = @dashboard-writable — can be set via dashboard, not via api
   * .why = declastruct tracks this for drift detection and dashboard guidance
   */
  registrantContact?: DeclaredCloudflareDomainRegistrantContact;

  /**
   * .what = transfer status indicators
   * .note = @dashboard-writable — transfer actions taken via dashboard
   * .why = declastruct tracks this for drift detection and transfer guidance
   */
  transferIn?: DeclaredCloudflareDomainTransferIn;
}

export class DeclaredCloudflareDomainRegistration
  extends DomainEntity<DeclaredCloudflareDomainRegistration>
  implements DeclaredCloudflareDomainRegistration
{
  public static primary = ['id'] as const;
  public static unique = ['name'] as const;
  public static metadata = ['id'] as const;
  public static readonly = [
    'available',
    'canRegister',
    'currentRegistrar',
    'expiresAt',
    'createdAt',
    'registryStatuses',
    'supportedTld',
  ] as const;
  public static nested = {
    registrantContact: DeclaredCloudflareDomainRegistrantContact,
    transferIn: DeclaredCloudflareDomainTransferIn,
  };
}
```

## 3. domain.operations

### 3.1 zone operations

| file | operation | description |
|------|-----------|-------------|
| `domainZone/getOneDomainZone.ts` | getOne | get zone by primary (id) or unique (name) |
| `domainZone/getAllDomainZones.ts` | getAll | list all zones in account |
| `domainZone/setDomainZone.ts` | findsert/upsert | create or update zone |
| `domainZone/delDomainZone.ts` | delete | delete a zone |
| `domainZone/castIntoDeclaredCloudflareDomainZone.ts` | cast | sdk response → domain object |

### 3.2 dns record operations

| file | operation | description |
|------|-----------|-------------|
| `domainDnsRecord/getOneDomainDnsRecord.ts` | getOne | get record by primary (id) or unique (zoneId, name, type) |
| `domainDnsRecord/getAllDomainDnsRecords.ts` | getAll | list all records in zone |
| `domainDnsRecord/setDomainDnsRecord.ts` | findsert/upsert | create or update record |
| `domainDnsRecord/delDomainDnsRecord.ts` | delete | delete a record |
| `domainDnsRecord/castIntoDeclaredCloudflareDomainDnsRecord.ts` | cast | sdk response → domain object |

### 3.3 domain registration operations

| file | operation | description |
|------|-----------|-------------|
| `domainRegistration/getOneDomainRegistration.ts` | getOne | get domain by unique (name) |
| `domainRegistration/getAllDomainRegistrations.ts` | getAll | list all domains in registrar |
| `domainRegistration/setDomainRegistration.ts` | findsert/upsert | update settings or guide registration |
| `domainRegistration/castIntoDeclaredCloudflareDomainRegistration.ts` | cast | sdk response → domain object |

**note**: `setDomainRegistration` has special behavior:
- if domain exists: update settings (autoRenew, privacy, locked)
- if domain does not exist: throw `HelpfulError` with registration guidance

## 6. guidance procedures (HelpfulErrors)

### 6.1 domain registration guidance

triggered when `setDomainRegistration.findsert` is called for non-existent domain:

```typescript
throw new HelpfulError({
  code: 'DOMAIN_REGISTRATION_REQUIRED',
  message: `domain "${domainName}" not found in cloudflare registrar`,
  guidance: {
    summary: 'domain registration requires manual action via cloudflare dashboard',
    steps: [
      '1. navigate to cloudflare dashboard → registrar → register domains',
      `2. search for domain: ${domainName}`,
      '3. select purchase and complete checkout',
      '4. verify domain appears in account',
      '5. re-run declastruct apply',
    ],
    link: 'https://dash.cloudflare.com/?to=/:account/domains/register',
  },
});
```

### 6.2 domain transfer guidance

triggered when `transferIn` status indicates pending steps:

```typescript
throw new HelpfulError({
  code: 'DOMAIN_TRANSFER_STEPS_REQUIRED',
  message: `domain "${domainName}" transfer requires completion`,
  guidance: {
    summary: 'domain transfer requires manual steps at both registrars',
    currentStatus: domain.transferIn,
    steps: [
      'at current registrar:',
      '  1. log into current registrar account',
      '  2. disable dnssec (note ds record ttl, wait for propagation)',
      '  3. disable whois privacy',
      '  4. unlock domain (remove clientTransferProhibited)',
      '  5. request authorization code',
      '',
      'at cloudflare:',
      '  6. navigate to dashboard → transfer domains',
      '  7. enter domain name and auth code',
      '  8. confirm contact information',
      '  9. complete transfer initiation',
      '',
      'post-initiation:',
      '  10. approve form of authorization (foa) email',
      '  11. wait for transfer completion (up to 5 days)',
      '  12. re-run declastruct apply',
    ],
    link: 'https://dash.cloudflare.com/?to=/:account/domains/transfer',
  },
});
```

### 6.3 zone activation guidance

triggered when `DomainZone.status === 'pending'`:

```typescript
throw new HelpfulError({
  code: 'ZONE_PENDING_ACTIVATION',
  message: `zone "${zoneName}" is pending activation`,
  guidance: {
    summary: 'zone pending activation - nameserver change required',
    steps: [
      `1. log into your domain registrar (${zone.originalRegistrar})`,
      '2. update nameservers to cloudflare\'s assigned servers:',
      ...zone.nameServers.map(ns => `     - ${ns}`),
      '3. wait for dns propagation (up to 24-48 hours)',
      '4. zone will auto-activate once verified',
      '5. re-run declastruct apply to continue',
    ],
    currentNameservers: zone.originalNameservers,
    requiredNameservers: zone.nameServers,
  },
});
```
