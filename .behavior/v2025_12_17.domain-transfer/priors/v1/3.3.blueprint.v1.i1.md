## 3. key implementation patterns

### 3.1 cast function pattern

```typescript
export const castIntoDeclaredCloudflareDomainZone = (
  input: Cloudflare.Zones.Zone,
): HasReadonly<typeof DeclaredCloudflareDomainZone> => {
  // fail-fast validation
  if (!input.id) UnexpectedCodePathError.throw('zone missing id', { input });
  if (!input.name) UnexpectedCodePathError.throw('zone missing name', { input });

  // cast with readonly enforcement
  return assure(
    DeclaredCloudflareDomainZone.as({
      id: input.id,
      name: input.name,
      type: input.type as 'full' | 'partial' | 'secondary',
      paused: input.paused,
      status: input.status,
      nameServers: input.name_servers,
      originalNameservers: input.original_name_servers,
      originalRegistrar: input.original_registrar,
      createdOn: input.created_on,
      activatedOn: input.activated_on,
    }),
    hasReadonly({ of: DeclaredCloudflareDomainZone }),
  );
};
```

### 3.2 getOne operation pattern

```typescript
export const getOneDomainZone = async (
  input: {
    by: PickOne<{
      primary: RefByPrimary<typeof DeclaredCloudflareDomainZone>;
      unique: RefByUnique<typeof DeclaredCloudflareDomainZone>;
    }>;
  },
  context: ContextCloudflareApi & VisualogicContext,
): Promise<HasReadonly<typeof DeclaredCloudflareDomainZone> | null> => {
  // handle by primary (id)
  if (input.by.primary) {
    try {
      const zone = await context.cloudflare.client.zones.get({
        zone_id: input.by.primary.id,
      });
      return castIntoDeclaredCloudflareDomainZone(zone);
    } catch (error) {
      if (error instanceof Cloudflare.NotFoundError) return null;
      throw error;
    }
  }

  // handle by unique (name)
  if (input.by.unique) {
    const { result } = await context.cloudflare.client.zones.list({
      name: input.by.unique.name,
    });
    if (!result.length) return null;
    return castIntoDeclaredCloudflareDomainZone(result[0]);
  }

  throw new UnexpectedCodePathError('invalid input', { input });
};
```

### 3.3 set operation pattern (with HelpfulError)

```typescript
export const setDomainRegistration = async (
  input: PickOne<{
    findsert: DeclaredCloudflareDomainRegistration;
    upsert: DeclaredCloudflareDomainRegistration;
  }>,
  context: ContextCloudflareApi & VisualogicContext,
): Promise<HasReadonly<typeof DeclaredCloudflareDomainRegistration>> => {
  const desired = input.findsert ?? input.upsert;
  if (!desired) throw new UnexpectedCodePathError('no input', { input });

  // lookup existing
  const found = await getOneDomainRegistration(
    { by: { unique: { name: desired.name } } },
    context,
  );

  // if not found, throw guidance error
  if (!found) {
    throw new HelpfulError({
      code: 'DOMAIN_REGISTRATION_REQUIRED',
      message: `domain "${desired.name}" not found in cloudflare registrar`,
      guidance: {
        summary: 'domain registration requires manual action via cloudflare dashboard',
        steps: [
          '1. navigate to cloudflare dashboard → registrar → register domains',
          `2. search for domain: ${desired.name}`,
          '3. select purchase and complete checkout',
          '4. verify domain appears in account',
          '5. re-run declastruct apply',
        ],
        link: 'https://dash.cloudflare.com/?to=/:account/domains/register',
      },
    });
  }

  // findsert returns existing
  if (input.findsert) return found;

  // upsert updates settings
  const updated = await context.cloudflare.client.registrar.domains.update(
    desired.name,
    {
      account_id: context.cloudflare.accountId,
      auto_renew: desired.autoRenew,
      locked: desired.locked,
      privacy: desired.privacy,
    },
  );
  return castIntoDeclaredCloudflareDomainRegistration(updated);
};
```

### 3.4 zone ref resolution pattern (for dns records)

```typescript
const resolveZoneRef = async (
  zoneRef: Ref<typeof DeclaredCloudflareDomainZone>,
  context: ContextCloudflareApi & VisualogicContext,
): Promise<string> => {
  // if primary ref (has id), use directly
  if ('id' in zoneRef && zoneRef.id) return zoneRef.id;

  // if unique ref (has name), resolve to id
  if ('name' in zoneRef && zoneRef.name) {
    const zone = await getOneDomainZone(
      { by: { unique: { name: zoneRef.name } } },
      context,
    );
    if (!zone) {
      throw new BadRequestError(`zone not found: ${zoneRef.name}`, { zoneRef });
    }
    return zone.id!;
  }

  throw new UnexpectedCodePathError('invalid zone ref', { zoneRef });
};
```
