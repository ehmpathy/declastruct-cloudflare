  when('zone does not exist', () => {
    then('creates the zone')
      sothat('users can onboard new domains')

  when('zone already exists', () => {
    then('returns the zone unchanged')
      sothat('operations are idempotent')
  })

      sothat('operations are ordered correctly')
  })

  when('record is deleted', () => {
    then('removes the record from the zone')
      sothat('users can clean up dns entries')
  })

  when('plan is applied', () => {
    then('creates zone first, then record')
      sothat('dependencies are resolved automatically')
  })
