# enframe.playloops: declastruct-cloudflare rapid feedback

## experience reproduction table

| experience | entry point | user actions | expected outcome | how to assert |
|------------|-------------|--------------|------------------|---------------|
| zone create | `setDomainZone({ findsert })` | declare zone, apply | zone created in cloudflare | `expect(zone.id).toBeDefined()` |
| zone lookup | `getOneDomainZone({ by })` | call with unique ref | zone returned or null | `expect(zone?.name).toEqual('example.com')` |
| dns record create | `setDomainDnsRecord({ findsert })` | declare record, apply | record created in zone | `expect(record.id).toBeDefined()` |
| dns record update | `setDomainDnsRecord({ upsert })` | modify content, apply | record updated | `expect(record.content).toEqual(newContent)` |
| dns record delete | `delDomainDnsRecord({ by })` | delete record | record removed | `expect(recordAfter).toBeNull()` |
| transfer guidance | `setDomainRegistration({ findsert })` | declare for unowned domain | BadRequestError with transfer steps | `expect(error.message).toContain('transfer')` |
| purchase guidance | `setDomainRegistration({ findsert })` | declare for unregistered domain | BadRequestError with purchase steps | `expect(error.message).toContain('purchase')` |
| registration update | `setDomainRegistration({ upsert })` | modify settings, apply | settings applied | `expect(reg.autoRenew).toBe(true)` |
| drift detection | `declastruct plan` | run plan command | changes shown in plan | `expect(plan.changes).toContainEqual(...)` |
| idempotent apply | `declastruct apply` twice | apply same plan twice | second apply succeeds | `expect(secondResult.errors).toHaveLength(0)` |

## playloop code sketches

### playloop.1: zone management

```typescript
import { given, when, then, useBeforeAll } from 'test-fns';
import { getContext } from '../.test/getSampleCloudflareApiContext';
import { getAllDomainZones, getOneDomainZone, setDomainZone } from '../domain.operations/domainZone';

describe('zone management', () => {
  given('[case1] a cloudflare account', () => {
    when('[t0] getAllDomainZones is called', () => {
      then('returns array of zones', async () => {
        const zones = await getAllDomainZones(getContext());
        expect(Array.isArray(zones)).toBe(true);
      });
    });

    when('[t1] getOneDomainZone is called for extant zone', () => {
      const zonesData = useBeforeAll(async () => {
        const allZones = await getAllDomainZones(getContext());
        return { firstZone: allZones[0] ?? null };
      });

      then('returns the zone', async () => {
        if (!zonesData.firstZone) return;  // skip if no zones
        const zone = await getOneDomainZone(
          { by: { unique: { name: zonesData.firstZone.name } } },
          getContext(),
        );
        expect(zone?.name).toEqual(zonesData.firstZone.name);
      });
    });
  });
});
```

### playloop.2: dns record management

```typescript
import { given, when, then, useBeforeAll } from 'test-fns';
import { RefByUnique } from 'domain-objects';
import { DeclaredCloudflareDomainZone } from '../domain.objects';
import { getAllDomainDnsRecords, setDomainDnsRecord } from '../domain.operations/domainDnsRecord';

describe('dns record management', () => {
  given('[case1] a zone that exists', () => {
    const scene = useBeforeAll(async () => {
      const zones = await getAllDomainZones(getContext());
      const zone = zones[0];
      if (!zone) return { zone: null };
      return { zone };
    });

    when('[t0] getAllDomainDnsRecords is called', () => {
      then('returns records for zone', async () => {
        if (!scene.zone) return;
        const records = await getAllDomainDnsRecords(
          { by: { zone: { name: scene.zone.name } } },
          getContext(),
        );
        expect(Array.isArray(records)).toBe(true);
      });
    });

    when('[t1] setDomainDnsRecord({ findsert }) is called', () => {
      then('creates or returns extant record', async () => {
        if (!scene.zone) return;
        const record = await setDomainDnsRecord(
          {
            findsert: {
              zone: RefByUnique.as<typeof DeclaredCloudflareDomainZone>({ name: scene.zone.name }),
              name: 'test.example.com',
              type: 'A',
              content: '127.0.0.1',
              ttl: 300,
            },
          },
          getContext(),
        );
        expect(record.id).toBeDefined();
      });
    });
  });
});
```

### playloop.3: transfer guidance

```typescript
import { given, when, then } from 'test-fns';
import { getError } from 'helpful-errors';
import { BadRequestError } from 'helpful-errors';
import { setDomainRegistration } from '../domain.operations/domainRegistration';

describe('registration guidance', () => {
  given('[case1] a domain registered elsewhere', () => {
    when('[t0] setDomainRegistration({ findsert }) is called', () => {
      then('throws BadRequestError with transfer guidance', async () => {
        const error = await getError(
          setDomainRegistration(
            {
              findsert: {
                name: 'google.com',  // registered elsewhere
                zone: { name: 'google.com' },
                autoRenew: true,
                locked: true,
                privacyProtection: true,
              },
            },
            getContext(),
          ),
        );
        expect(error).toBeInstanceOf(BadRequestError);
        expect(error.message).toContain('transfer');
      });
    });
  });
});
```

### playloop.4: acceptance test via CLI

```typescript
import { execSync } from 'child_process';
import { existsSync, readFileSync, mkdirSync } from 'fs';
import { join } from 'path';
import { given, when, then, useBeforeAll } from 'test-fns';

describe('declastruct CLI workflow', () => {
  const repoRoot = join(__dirname, '..', '..', '..');
  const testDir = join(__dirname, '.test', '.temp', 'acceptance');
  const resourcesFile = join(__dirname, '.test', 'assets', 'resources.acceptance.ts');
  const planFile = join(testDir, 'plan.json');

  beforeAll(() => mkdirSync(testDir, { recursive: true }));

  given('[case1] a declastruct resources file', () => {
    when('[t0] a plan is generated via CLI', () => {
      const prep = useBeforeAll(async () => {
        execSync(
          `npx declastruct plan --wish ${resourcesFile} --into ${planFile}`,
          { stdio: 'inherit', env: process.env, cwd: repoRoot },
        );
        return { plan: JSON.parse(readFileSync(planFile, 'utf-8')) };
      });

      then('creates valid plan file', () => {
        expect(existsSync(planFile)).toBe(true);
        expect(prep.plan).toHaveProperty('changes');
        expect(Array.isArray(prep.plan.changes)).toBe(true);
      });
    });

    when('[t1] plan is applied via CLI', () => {
      then('apply succeeds', () => {
        const result = execSync(
          `npx declastruct apply --plan ${planFile}`,
          { env: process.env, cwd: repoRoot },
        ).toString('utf-8');
        expect(result).toContain('applied');
      });
    });

    when('[t2] same plan is applied again (idempotency)', () => {
      then('second apply succeeds', () => {
        const result = execSync(
          `npx declastruct apply --plan ${planFile}`,
          { env: process.env, cwd: repoRoot },
        ).toString('utf-8');
        expect(result).not.toContain('error');
      });
    });
  });
});
```

## playloop benefits

1. **reproduces real user journey in seconds** — integration tests call same sdk methods users call
2. **friction felt in playloop = friction felt by user** — if api is awkward, test is awkward
3. **defects caught before user sees them** — acceptance tests validate CLI workflow end-to-end
